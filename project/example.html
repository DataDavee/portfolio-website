4. SQL Showcase: Data Cleaning & Feature Engineering Examples

Example 1: Handling Duplicate Records


-- Identify potential duplicate sales based on address, sale date, and price
WITH RankedSales AS (
    SELECT
        SaleID,
        PropertyAddress,
        SaleDate,
        SalePrice,
        ROW_NUMBER() OVER (PARTITION BY PropertyAddress, SaleDate, SalePrice ORDER BY SaleID) as rn
    FROM
        RawRealEstateTransactions
)
SELECT
    SaleID,
    PropertyAddress,
    SaleDate,
    SalePrice
FROM
    RankedSales
WHERE
    rn > 1;

Example 2: Feature Engineering - Calculating Price Per Square Foot Safely


-- Calculate PricePerSqFt, handling cases where LivingAreaSqFt might be zero or null
SELECT
    PropertyID,
    SalePrice,
    LivingAreaSqFt,
    CASE
        WHEN LivingAreaSqFt IS NOT NULL AND LivingAreaSqFt > 0
        THEN CAST(SalePrice AS DECIMAL(18, 2)) / LivingAreaSqFt
        ELSE 0.00 -- Assign 0 or NULL for invalid living areas, depending on requirement
    END AS PricePerSqFt
FROM
    CleanedRealEstateData;

Example 3: Standardizing Property Types and Handling Missing Values


-- Standardize various property type entries and fill missing values
UPDATE CleanedRealEstateData
SET PropertyType =
    CASE
        WHEN PropertyType IS NULL OR TRIM(PropertyType) = '' THEN 'Unknown'
        WHEN PropertyType LIKE '%Single Family%' OR PropertyType LIKE '%SFH%' THEN 'Single Family Home'
        WHEN PropertyType LIKE '%Condo%' OR PropertyType LIKE '%Cndo%' THEN 'Condominium'
        WHEN PropertyType LIKE '%Multi-Family%' OR PropertyType LIKE '%MFA%' THEN 'Multi-Family Dwelling'
        WHEN PropertyType LIKE '%Townhouse%' THEN 'Townhouse'
        ELSE 'Other Residential' -- Categorize any remaining types
    END;
